На самом деле исторически все не совсем так.

backbone.js - легковесный фреймворк, созданный в 2010-м году который стал очень быстро популярен в противовес монстрам типа ExtJS. Он активно применяется такими игроками как Pinterest и AirBNB.

angular.js - Создан в 2009 (кризис), создатели переписали приложение из 17к строк кода которое писали пол-года до 1к за 3 недели чем сильно привлекли Google. Которая и взяла фреймворк под свое крыло. Актовно используется в Google и Virgin.

ember.js - С 2007-го года писался разными людьми и компаниями, побывала там даже Apple. В конце кноцов был фокркнут core contributor-ом таких проектов как Rails и jQuery Yehuda Katz. Используется Yahoo!, Groupon.

backbone.js

  • +Структурирует код;
  • +Давно популярен;
  • +Минималистичный;
  • +Свобода решений;
  • -Свобода решений;
  • -Много кода.

Структурирует код - история про iMallTool.

angular.js

  • +Быстрое прототипирование;
  • +Повышенное внимание тестированию;
  • +Easy to learn, difficult to master;
  • +Правильное место для HTML во View;
  • -Скорость/Расточительность;
  • -Весь в себе;
  • -“Стихийный” дизайн.

Скорость/Расточительность - медленный watcher.

“Стихийный” дизайн - difficult to master начинается слишком рано.

Очень быстрый - история про tForm.

$(document).ready(function() {
  var Currency = Backbone.Model.extend({
    defaults: {
      dollars: 0
    },
    hryvnias: function(value) {
      return this.get('dollars') * kurs;
    }
  })

  var ConverterView = Backbone.View.extend({
    el: document.getElementById('backbone'),
    model: new Currency(),
    events: {
      'input #backboneDollar': 'updateDollars',
    },
    initialize: function() {
      this.listenTo(this.model, 'change:dollars', this.render)
      this.render()
    },
    render: function() {
      this.$('#backboneDollar').val(this.model.get('dollars'))
      this.$('#backboneHryvnia').val(this.model.hryvnias())
    },
    updateDollars: function(event) {
      this.model.set('dollars', event.target.value)
    },
  })

  var converterView = new ConverterView();
});
